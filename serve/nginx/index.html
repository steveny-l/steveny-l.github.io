Nginx

Install

Env

CentOS 7.4

Install make

  yum -y install gcc automake autoconf libtool make

Install g++

  yum -y install gcc gcc-c++

Install pcre

Check pcre version

ftp://ftp.csx.cam.ac.uk/pub/software/programming/pcre/

  cd /usr/local/src

  wget ftp://ftp.csx.cam.ac.uk/pub/software/programming/pcre/pcre-8.43.tar.gz

  tar -zxvf pcre-8.43.tar.gz && cd pcre-8.43

  ./configure && make && make install

Install zlib

Check zlib version

http://www.zlib.net/

  cd /usr/local/src

  wget http://zlib.net/zlib-1.2.11.tar.gz

  tar -zxvf zlib-1.2.11.tar.gz && cd zlib-1.2.11

  ./configure && make && make install

Install openssl

Check openssl version

ftp://ftp.openssl.org/source/

  cd /usr/local/src

  wget http://www.openssl.org/source/openssl-1.0.2r.tar.gz

  tar -zxvf openssl-1.0.2r.tar.gz && cd openssl-1.0.2r

  ./config && make && make install

Install nginx

Check nginx version

http://nginx.org/en/download.html

  cd /usr/local/src

  wget http://nginx.org/download/nginx-1.14.2.tar.gz

  tar -zxvf nginx-1.14.2.tar.gz && cd nginx-1.14.2

  ./configure --sbin-path=/usr/local/nginx/nginx \--conf-path=/usr/local/src/nginx/nginx.conf \--pid-path=/usr/local/src/nginx/nginx.pid \--with-http_ssl_module \--with-pcre=/usr/local/src/pcre-8.43 \--with-zlib=/usr/local/src/zlib-1.2.11 \--with-openssl=/usr/local/src/openssl-1.0.2r

  make && make install

  /usr/sbin/groupadd www

  /usr/sbin/useradd -g www www

Run nginx

  /usr/sbin/groupadd www

  /usr/sbin/useradd -g www www

  /usr/local/nginx/nginx
